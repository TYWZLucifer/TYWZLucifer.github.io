JS中的变量类型与强制转换

JS中一共有六种数据类型：

String 字符串 Number 数值 Boolean 布尔值 Null 空值 Undefined 未定义 Object 对象

其中，前五个都属于基本数据类型，而Object属于引用数据类型。

String类型

var a = "abc";
console.log(typeof a);  //string
a = "这是一个双引号 \"";
console.log(a);
a = "这是一个换行符 \n";
console.log(a);
a = "这是一个制表符 \t";
console.log(a);

Number类型

var a = 123;
console.log(typeof a);  //number
a = "abc" * "def";
console.log(a); //NaN Not a Number
a = NaN;
console.log(typeof a);  //NaN也是number类型
a = 0.1 + 0.2;
console.log(a); //0.30000000000000004 JS中浮点类型并不精确
a = Number.MAX_VALUE+Number.MAX_VALUE;
console.log(a); //Infinity表示超过了number的最大表示范围
a = Infinity;
console.log(typeof a);  //Infinity也是number类型

Boolean类型

布尔值只有true和false两个值。

Null和Undefined类型

null专门用来表示一个空的对象，所以它的类型不是Null。undefined只用来表示声明的未定义的变量。

var a = null;
console.log(typeof a); //Object
var b;
console.log(b); //Undefined

强制转换为String类型

//调用被转数据类型的toString()方法，但不会改变原数据的类型，注意Null和Undefined是没有toString()方法的
var a = 123;
console.log(typeof a); //number
var b = a.toString(); 
console.log(typeof a); //number
console.log(typeof b); //string
a = null;
b = a.toString(); //无效
a = undefined;
b = a.toString(); //无效

//调用String()函数,将转换变量作为参数传入，对于Number和Boolean实际还是调用toString()，但对null和undefined可以转为字符串
a = 123;
a = String(a);
console.log(a); //string
a = null;
a = String(a);
console.log(a); //null
console.log(typeof a); //string
a = undefined;
a = String(a);
console.log(a); //undefined
console.log(typeof a); //string

强制转换为Number类型

// 调用Number()函数
var a = "123";
a = Number(a);
console.log(a); //123
console.log(typeof a); //number
a = "abc"; //abc不能转换为数字
a = Number(a);
console.log(a); //NaN
console.log(typeof a); //number
a = "  ";
a = Number(a); //如果字符串是空串或者全是空格就会转为0
console.log(a); //0
console.log(typeof a); //number
a = true;
a = Number(a); //布尔值真为1
console.log(a); //1
a = false;
a = Number(a); //布尔值假为0
console.log(a); //0
a = null;
a = Number(a); //null转为数字就是0
console.log(a);
a = undefined;
a = Number(a);
console.log(a); // undefined只能转为NaN

// 专门负责字符串的转换:parseInt()--字符串转整数 parseFloat()字符串转浮点数
a = "123xp";
a = parseInt(a);
console.log(a); //123 parseInt()将字符串中有效的数字部分取出转为int
a = "123def567";
a = parseInt(a);
console.log(a); //123
a = "gg123def567";
a = parseInt(a);
console.log(a); //NaN
//对于非string类型的变量如a=true，会先转为字符串“true”，再作为参数，所以返回是NaN

强制转换为布尔类型

对于数字转布尔，除了0和NaN为false，其它值都是true。

对于字符串转布尔，除了空串为false，其余都是true，包括空格串。

Null转布尔是false。

Undefined转布尔是false。

使用算术运算符转换数据类型

// 对非Number类型的数据运算（加法对字符串有特殊）时，会将其先转为Number再运算，所以对直接a-0可以转number
var a = "123" - 0;
console.log(typeof a); //number
a = +"123"
console.log(typeof a); //number 正负号直接将数据转为Number
a = 1 + true;
console.log(typeof a); //number
a = a + NaN;
console.log(a); //NaN 任何值与NaN运算都是NaN
// 两个字符串相加不是Number，而是拼接字符串，注意只有加法！！
a = "123" + "456";
console.log(a); // 123456
console.log(typeof a); //string
// 任何值与字符串相加都是先转为字符串，然后拼串
a = 123 + "456";
console.log(a); //123456
// 所以对数据直接加一个“”空串可以转为字符串
