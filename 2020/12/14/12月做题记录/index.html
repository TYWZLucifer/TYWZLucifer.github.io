字母异位词分组

用单词的字典序做key，更秀的是JS中object可以用数组做key，可以统计26个字母的出现次数

var groupAnagrams = function(strs) {
    const map = new Map();
    for(let str of strs) {
        let array = Array.from(str);
        array.sort();
        let key = array.toString();
        let list = map.get(key)?map.get(key):new Array();
        list.push(str);
        map.set(key,list);
    }
    return Array.from(map.values());
};

var groupAnagrams = function(strs) {
    const map = new Object();
    for(let s of strs) {
        const count = new Array(26).fill(0);
        for(let c of s) count[c.charCodeAt()-'a'.charCodeAt()]++;
        map[count]?map[count].push(s):map[count]=[s];
    }
    return Object.values(map);
};

单调递增的数字

var monotoneIncreasingDigits = function(N) {
    const strN = N.toString().split('').map(v => +v); //将V变成number类型
    let i = 1;
    while(i<strN.length && strN[i-1]<=strN[i]) i++;
    if(i < strN.length) {
        while(i>0 && strN[i-1]>strN[i]) {
            strN[i-1]--;
            i--;
        }
        for(i++;i<strN.length;i++) strN[i]=9;
    }
    return parseInt(strN.join(''));
};

买卖股票的最佳时机含手续费

DP和贪心都可以

var maxProfit = function(prices, fee) {
    let n = prices.length;
    let buy = prices[0]+fee;
    let profit = 0;
    for(let i=1;i<n;i++) {
        if(prices[i]+fee < buy) buy = prices[i]+fee;
        else if(prices[i] > buy) {
            profit += prices[i] - buy;
            buy = prices[i];
        }
    }
    return profit;
};

找不同

用排序做时间复杂度O(logN)，可以遍历O(N)解决

var findTheDifference = function(s, t) {
    let sarr = s.split('').sort();
    let tarr = t.split('').sort();
    for(let i=0;i<sarr.length;i++) {
        if(sarr[i] !== tarr[i]) return tarr[i];
    }
    return tarr[sarr.length];
};

var findTheDifference = function(s, t) {
    let res = 0;
    for(const ch of s) res ^= ch.charCodeAt();
    for(const ch of t) res ^= ch.charCodeAt();
    return String.fromCharCode(res);
};

旋转图像

水平翻转加对角线反转就是旋转

var rotate = function(matrix) {
    const n = matrix.length;
    for(let i = 0;i < Math.floor(n/2);i++) {
        for(let j = 0;j < n;j++) {
            [matrix[i][j], matrix[n-i-1][j]] = [matrix[n-i-1][j], matrix[i][j]];
        }
    }
    for(let i = 0;i < n;i++) {
        for(let j = 0;j < i;j++) {
            [matrix[i][j], matrix[j][i]] = [matrix[j][i], matrix[i][j]];
        }
    }
};

去除重复字母

indexOf从i开始找栈顶元素保证栈中可以保留每个字符一次，不会误删除

var removeDuplicateLetters = function(s) {
    let stack = [];
    for(let i = 0, len = s.length;i < len;i++) {
        let ch = s[i];
        if(stack.indexOf(ch) > -1) continue;
        while(stack.length > 0 && stack[stack.length - 1] > ch && s.indexOf(stack[stack.length - 1], i) > i) stack.pop();
        stack.push(ch);
    }
    return stack.join('');
};

二叉树的锯齿形层序遍历

用双端队列写更简单，一层循环用布尔值控制方向(push还是unshift)

var zigzagLevelOrder = function(root) {
    if(root === null) return [];
    let left = [];
    let right = [];
    let res = new Array();
    let i = 0;
    left.push(root);
    while(left.length !== 0 || right.length !== 0) {
        res[i] = new Array();
        if(left.length !== 0) {
            while(left.length !== 0) {
                let node = left.pop();
                res[i].push(node.val);
                if(node.left !== null) right.push(node.left);
                if(node.right !== null) right.push(node.right);
            }
            i++;
            continue;
        }
        if(right.length !== 0) {
            while(right.length !== 0) {
                let node = right.pop();
                res[i].push(node.val);
                if(node.right !== null) left.push(node.right);
                if(node.left !== null) left.push(node.left);
            }
            i++;
        }
    }
    return res;
};

使用最小花费爬楼梯

到达i的方式只有两种：从i-1走一步或从i-2走两步，动态规划即可，可以优化空间

var minCostClimbingStairs = function(cost) {
    let dp = new Array(cost.length).fill(0);
    dp[0] = cost[0];
    dp[1] = cost[1];
    for(let i = 2, len = cost.length;i < len;i++) {
        dp[i] = cost[i] + Math.min(dp[i-1],dp[i-2]);
    }
    return Math.min(dp[cost.length-1],dp[cost.length-2]);
};

分发糖果

同时满足左遍历和右遍历条件的最大值，没太看懂如何证明有效

var candy = function(ratings) {
    let left = new Array(ratings.length).fill(1);
    let right = new Array(ratings.length).fill(1);
    for(let i = 1, len = ratings.length;i < len;i++) {
        if(ratings[i] > ratings[i-1]) left[i] = left[i-1] + 1;
    }
    let count = left[ratings.length - 1];
    for(let i = ratings.length - 2;i >= 0;i--) {
        if(ratings[i] > ratings[i+1]) right[i] = right[i+1] + 1;
        count += Math.max(left[i],right[i]);
    }
    return count;
};

分发饼干

贪心就可以

var findContentChildren = function(g, s) {
    g.sort((a, b) => a - b);
    s.sort((a, b) => a - b);
    let childNum = g.length, cookieNum = s.length;
    let res = 0;
    for(let i = 0, j = 0;i < childNum && j < cookieNum;i++, j++) {
        while(j < cookieNum && g[i] > s[j]) j++;
        if(j < cookieNum) res++;
    }
    return res;
};

最大矩形

和之前的最大柱状图一个道理，统计矩阵每个点左侧1的数量后就相当于将每一列都看作一排柱状图的底，将这些列作为输入送到最大柱状图的代码中找到最大值即可

var maximalRectangle = function(matrix) {
    const m = matrix.length;
    if (m === 0) {
        return 0;
    }
    const n = matrix[0].length;
    const left = new Array(m).fill(0).map(() => new Array(n).fill(0));

    for (let i = 0; i < m; i++) {
        for (let j = 0; j < n; j++) {
            if (matrix[i][j] === '1') {
                left[i][j] = (j === 0 ? 0 : left[i][j - 1]) + 1;
            }
        }
    }

    let height = new Array(m);
    let res = 0;
    for(let i = 0; i < n; i++) {
        for(let j = 0; j < m; j++) {
            height[j] = left[j][i];
        }
        res = Math.max(res,largestRectangleArea(height));
    }

    return res;
};

var largestRectangleArea = function(heights) {
    let maxArea = 0;
    const stack = [];
    heights = [0,...heights,0];
    for(let i=0;i<heights.length;i++) {
        while(heights[i] < heights[stack[stack.length-1]]) {
            let stackTopIndex = stack.pop();
            maxArea = Math.max(maxArea,heights[stackTopIndex]*(i-stack[stack.length-1]-1));
        }
        stack.push(i);
    }
    return maxArea;
};
